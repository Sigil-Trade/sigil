use anchor_lang::prelude::*;

use crate::errors::AgentShieldError;
use crate::events::OracleRegistryUpdated;
use crate::state::*;

#[derive(Accounts)]
pub struct ApplyOracleUpdate<'info> {
    #[account(mut)]
    pub authority: Signer<'info>,

    #[account(
        mut,
        seeds = [b"oracle_registry"],
        bump,
    )]
    pub oracle_registry: AccountLoader<'info, OracleRegistry>,

    #[account(
        mut,
        seeds = [b"pending_oracle_update"],
        bump = pending_oracle_update.bump,
        close = authority,
    )]
    pub pending_oracle_update: Account<'info, PendingOracleUpdate>,
}

pub fn handler(ctx: Context<ApplyOracleUpdate>) -> Result<()> {
    let mut registry = ctx.accounts.oracle_registry.load_mut()?;

    // NOTE: We check the CURRENT registry authority, not the authority who
    // queued this update. This is deliberate — after authority transfer,
    // the new authority can apply or cancel pending updates. The new
    // authority can always cancel via cancel_oracle_update if they disagree.
    require!(
        registry.authority == ctx.accounts.authority.key(),
        AgentShieldError::UnauthorizedRegistryAdmin
    );

    // Timelock check
    let clock = Clock::get()?;
    let pending = &ctx.accounts.pending_oracle_update;
    require!(
        pending.is_ready(clock.unix_timestamp),
        AgentShieldError::TimelockNotExpired
    );

    let mut count = registry.count as usize;

    // Apply removals (swap-remove for O(1) per removal)
    let mut removed_count: u16 = 0;
    for mint in &pending.mints_to_remove {
        let mut i = 0;
        while i < count {
            if registry.entries[i].mint == *mint {
                count -= 1;
                if i < count {
                    registry.entries[i] = registry.entries[count];
                }
                registry.entries[count] = OracleEntryZC::default();
                removed_count = removed_count.saturating_add(1);
            } else {
                i += 1;
            }
        }
    }

    // Apply additions (update existing duplicates)
    let mut added_count: u16 = 0;
    for entry in &pending.entries_to_add {
        let mut found = false;
        for i in 0..count {
            if registry.entries[i].mint == entry.mint {
                registry.entries[i] = OracleEntryZC::from(entry);
                found = true;
                break;
            }
        }
        if !found {
            require!(
                count < MAX_ORACLE_ENTRIES,
                AgentShieldError::OracleRegistryFull
            );
            registry.entries[count] = OracleEntryZC::from(entry);
            count += 1;
            added_count = added_count.saturating_add(1);
        }
    }

    registry.count = count as u16;

    emit!(OracleRegistryUpdated {
        added_count,
        removed_count,
        total_entries: registry.count,
    });

    Ok(())
}
