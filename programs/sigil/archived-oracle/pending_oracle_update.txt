use super::MAX_ORACLE_UPDATE_BATCH;
use crate::state::OracleEntry;
use anchor_lang::prelude::*;

/// Queued oracle registry update that becomes executable after a 24h
/// timelock. Singleton — only one pending update at a time. Created by
/// `queue_oracle_update`, applied by `apply_oracle_update`, or cancelled
/// by `cancel_oracle_update`.
///
/// PDA seeds: `[b"pending_oracle_update"]`
#[account]
pub struct PendingOracleUpdate {
    /// Unix timestamp when this update was queued
    pub queued_at: i64,

    /// Unix timestamp when this update becomes executable
    pub executes_at: i64,

    /// Authority who queued this update
    pub authority: Pubkey,

    /// Oracle entries to add (max 10 per batch)
    pub entries_to_add: Vec<OracleEntry>,

    /// Token mints to remove from the registry (max 10 per batch)
    pub mints_to_remove: Vec<Pubkey>,

    /// Bump seed for PDA
    pub bump: u8,
}

impl PendingOracleUpdate {
    /// Worst-case size with all vectors at max capacity.
    ///
    /// discriminator (8) + queued_at (8) + executes_at (8) +
    /// authority (32) +
    /// entries_to_add: (4 + 97 * 10) = 974 +
    /// mints_to_remove: (4 + 32 * 10) = 324 +
    /// bump (1) = 1,355 bytes
    pub const SIZE: usize = 8
        + 8 // queued_at
        + 8 // executes_at
        + 32 // authority
        + (4 + OracleEntry::SIZE * MAX_ORACLE_UPDATE_BATCH) // entries_to_add
        + (4 + 32 * MAX_ORACLE_UPDATE_BATCH) // mints_to_remove
        + 1; // bump

    /// Returns true if the timelock period has expired and the update
    /// can be applied.
    pub fn is_ready(&self, current_timestamp: i64) -> bool {
        current_timestamp >= self.executes_at
    }
}
