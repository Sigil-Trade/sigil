use anchor_lang::prelude::*;

use crate::errors::AgentShieldError;
use crate::events::OracleUpdateQueued;
use crate::state::*;

#[derive(Accounts)]
pub struct QueueOracleUpdate<'info> {
    #[account(mut)]
    pub authority: Signer<'info>,

    #[account(
        seeds = [b"oracle_registry"],
        bump,
    )]
    pub oracle_registry: AccountLoader<'info, OracleRegistry>,

    #[account(
        init,
        payer = authority,
        space = PendingOracleUpdate::SIZE,
        seeds = [b"pending_oracle_update"],
        bump,
    )]
    pub pending_oracle_update: Account<'info, PendingOracleUpdate>,

    pub system_program: Program<'info, System>,
}

pub fn handler(
    ctx: Context<QueueOracleUpdate>,
    entries_to_add: Vec<OracleEntry>,
    mints_to_remove: Vec<Pubkey>,
) -> Result<()> {
    let registry = ctx.accounts.oracle_registry.load()?;

    // Authority check
    require!(
        registry.authority == ctx.accounts.authority.key(),
        AgentShieldError::UnauthorizedRegistryAdmin
    );
    drop(registry);

    // Batch size bounds
    require!(
        entries_to_add.len() <= MAX_ORACLE_UPDATE_BATCH,
        AgentShieldError::OracleUpdateBatchTooLarge
    );
    require!(
        mints_to_remove.len() <= MAX_ORACLE_UPDATE_BATCH,
        AgentShieldError::OracleUpdateBatchTooLarge
    );

    // At least one operation
    require!(
        !entries_to_add.is_empty() || !mints_to_remove.is_empty(),
        AgentShieldError::OracleUpdateBatchTooLarge
    );

    let clock = Clock::get()?;
    let executes_at = clock
        .unix_timestamp
        .checked_add(ORACLE_UPDATE_TIMELOCK)
        .ok_or(AgentShieldError::Overflow)?;

    let pending = &mut ctx.accounts.pending_oracle_update;
    pending.queued_at = clock.unix_timestamp;
    pending.executes_at = executes_at;
    pending.authority = ctx.accounts.authority.key();
    pending.entries_to_add = entries_to_add.clone();
    pending.mints_to_remove = mints_to_remove.clone();
    pending.bump = ctx.bumps.pending_oracle_update;

    emit!(OracleUpdateQueued {
        executes_at,
        add_count: entries_to_add.len() as u16,
        remove_count: mints_to_remove.len() as u16,
    });

    Ok(())
}
