use anchor_lang::prelude::*;

use crate::errors::AgentShieldError;
use crate::events::OracleUpdateCancelled;
use crate::state::*;

#[derive(Accounts)]
pub struct CancelOracleUpdate<'info> {
    #[account(mut)]
    pub authority: Signer<'info>,

    #[account(
        seeds = [b"oracle_registry"],
        bump,
    )]
    pub oracle_registry: AccountLoader<'info, OracleRegistry>,

    #[account(
        mut,
        seeds = [b"pending_oracle_update"],
        bump = pending_oracle_update.bump,
        close = authority,
    )]
    pub pending_oracle_update: Account<'info, PendingOracleUpdate>,
}

pub fn handler(ctx: Context<CancelOracleUpdate>) -> Result<()> {
    let registry = ctx.accounts.oracle_registry.load()?;

    // Authority check
    require!(
        registry.authority == ctx.accounts.authority.key(),
        AgentShieldError::UnauthorizedRegistryAdmin
    );

    emit!(OracleUpdateCancelled {});

    Ok(())
}
